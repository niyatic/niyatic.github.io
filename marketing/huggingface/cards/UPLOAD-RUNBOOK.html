Step 2 (alt) — Python huggingface_hub equivalent

# Equivalent to Step 2; run only after the same human go. Token is read from the
# environment (set via `source huggingface.env`), never hard-coded or printed.
import os
from huggingface_hub import HfApi
api = HfApi(token=os.environ["HF_TOKEN"])  # loaded from huggingface.env; not echoed
CARDS = "/Users/niyati/Desktop/hyperapi-gtm-hub/huggingface/cards"
HF_ORG = "<HF_ORG>"
uploads = [
    ("MODEL-CARD-Qwen2.5-VL-7B-F.md",        f"{HF_ORG}/Qwen2.5-VL-7B-F",     "model"),
    ("MODEL-CARD-Qwen2.5-VL-3B-F.md",        f"{HF_ORG}/Qwen2.5-VL-3B-F",     "model"),
    ("MODEL-CARD-hyperapi-parse-intent.md",  f"{HF_ORG}/hyperapi-parse-intent","model"),
    ("DATASET-CARD-savior-bench-v1.md",      f"{HF_ORG}/savior-bench-v1",     "dataset"),
]
for local, repo_id, repo_type in uploads:
    api.upload_file(
        path_or_fileobj=f"{CARDS}/{local}",
        path_in_repo="README.md",
        repo_id=repo_id,
        repo_type=repo_type,
        commit_message=f"Add card ({local}) — cards-without-weights; W5-approved",
    )
